Here's a simple Python function that demonstrates a basic web scraper:

import requests
from bs4 import BeautifulSoup

def scrape_webpage(url):
    """
    Simple function to scrape a webpage
    """
    # Send request to the URL
    response = requests.get(url)
    
    # Check if the request was successful
    if response.status_code == 200:
        # Parse the HTML content
        soup = BeautifulSoup(response.text, 'html.parser')
        
        # Extract the title
        title = soup.title.string
        
        # Extract all paragraphs
        paragraphs = [p.text for p in soup.find_all('p')]
        
        return {
            'title': title,
            'paragraphs': paragraphs
        }
    
    return None

# Example usage
if __name__ == "__main__":
    data = scrape_webpage('https://example.com')
    print(f"Title: {data['title']}")
    print(f"Found {len(data['paragraphs'])} paragraphs")

Here's a JavaScript example for creating a simple dark mode toggle:

// Simple dark mode toggle functionality
function toggleDarkMode() {
    const body = document.body;
    const isDarkMode = body.classList.contains('dark-mode');
    
    // Toggle dark mode class
    if (isDarkMode) {
        body.classList.remove('dark-mode');
        localStorage.setItem('theme', 'light');
    } else {
        body.classList.add('dark-mode');
        localStorage.setItem('theme', 'dark');
    }
}

// Check user preference on page load
document.addEventListener('DOMContentLoaded', () => {
    const savedTheme = localStorage.getItem('theme');
    const prefersDark = window.matchMedia('(prefers-color-scheme: dark)').matches;
    
    if (savedTheme === 'dark' || (!savedTheme && prefersDark)) {
        document.body.classList.add('dark-mode');
    }
    
    // Add event listener to toggle button
    const toggleButton = document.getElementById('dark-mode-toggle');
    if (toggleButton) {
        toggleButton.addEventListener('click', toggleDarkMode);
    }
});
